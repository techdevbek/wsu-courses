\centerline{HOMEWORK I - MATH 420}
\begin{enumerate}
\item
A real-valued function $f$ defined on the real line is called an even function if $f(t)=f(-t)$, for each real number $t$. Prove that the set of even functions defined on the real line with the usual operations of addition and scalar multiplication is a vector space.

\item
Let $V$ denote the set of ordered pairs of real numbers. If $(x_1, x_2)\in V$, $(y_1, y_2)\in V$, and $c\in\mathbb{F}$, define
$$(x_1, x_2)+(y_1, y_2)=(x_1+y_1,x_1y_1) \mbox{ and } c(x_1, x_2)=(cx_1, cx_2)$$

Is $V$ a vector space over $\mathbb{R}$?
\item Prove that the set of $n\times n$ upper triangular matrices is a subspace of the $\mathcal{M}_{n\times n}(\mathbb{R})$.

 \item Let $U$ and $W$ be subspaces of a vector space $V$. Prove that $U\cup W$ is a subspace of $V$ if and only if $U\subseteq W$ or $W\subseteq U$.
     \end{enumerate}
