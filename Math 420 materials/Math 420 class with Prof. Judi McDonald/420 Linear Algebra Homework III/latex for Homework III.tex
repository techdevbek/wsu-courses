For each part, determine if the set $S$ is linearly independent.
\begin{enumerate}
\item $S=\{\left[\begin{array}{cc}
1&0\\-2&1\end{array}\right],\ \left[\begin{array}{cc}
0&-1\\1&1\end{array}\right], \ \left[ \begin{array}{cc}
-1&2\\1&0\end{array}\right]
, \ \left[ \begin{array}{cc}
2&1\\-4&4\end{array}\right]\}$
\item $S=\{\left[\begin{array}{cc}
1&0\\-2&1\end{array}\right],\ \left[\begin{array}{cc}
0&-1\\1&1\end{array}\right], \ \left[ \begin{array}{cc}
-1&2\\1&0\end{array}\right]
, \ \left[ \begin{array}{cc}
2&1\\2&-2\end{array}\right]\}$
\item $S=\{x^3+2x^2, -x^2+3x+1,x^3-x^2+2x-1\}$
\item  $S=\{x^3-x, 2x^2+4,-2x^3+3x^2+2x+6\}$
\end{enumerate}

\item Let $S$ be a set of nonzero polynomials in $\mathbb{P}$ such that no two have the same degree. Prove that $S$ is linearly independent.


\item Prove that if $S$ is a linearly independent subset of $M_{n\times n}(\mathbb{R})$, then $S^T=\{A^T|A\in S\}$ is also linearly independent.

 \item Let $S_1$  and $S_2$
 be disjoint linearly independent subsets of a vector space $V$. Prove that
$S_1\cup S_2$ is linearly dependent if and only if $$\mbox{Span }(S_1)\cap\mbox{ Span} (S_2)\neq \{0\}$$.


     \end{enumerate}
