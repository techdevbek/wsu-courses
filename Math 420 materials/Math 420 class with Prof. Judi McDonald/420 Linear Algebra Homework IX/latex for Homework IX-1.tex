\begin{enumerate}

\item Let $T:\mathbb{R}^4\rightarrow\mathbb{R}^4$ be given by
$$T\left(\left[\begin{array}{cccc}
a\\b\\c\\d\end{array}\right]\right)=\left[\begin{array}{cccc}
a+b\\b+c\\a+c\\a+d\end{array}\right]$$
Let $W$ be the $T$cyclic generated by $e_1$, the first vector of the standard basis.
\begin{enumerate}
\item Find a basis for $W$.
\item Find the characteristic polynomial of $T|_W$.
\item Find a basis for the $T$-cyclic subspace generated by $e_4$, the fourth vector of the standard basis.
\item Find the characteristic polynomial of $T$.
\item Verify that the characteristic polynomial of $T|_W$ divides the characteristic polynomial of $T$.
\end{enumerate}



\item Let $T$ be a linear operator on a vector space with a $T$-invariant subspace $W$. Prove that if $v$ is an eigenvector of
$T|_W$ with corresponding eigenvalue $\lambda$, then $v$ is also an eigenvector of $T$ with corresponding eigenvalue $\lambda$.


\item Let $T$ be a linear operator on a vector space $V$, let $v$ be a nonzero vector in $V$, and let $W$ be the $T$-cyclic subspace of $V$ generated by $v$. For any $u\in V$ , prove that $u\in W$  if and only if there exists a polynomial $g(t)$ such that $u=g(T)v$ .

\item Let $T$ be a linear operator on a finite-dimensional vector space $V$, and let $W$ be a $T$-invariant subspace of $V$. Suppose that
$v_1,\ v_2,\ldots,\ v_k$ are eigenvectors of $T$ corresponding to distinct eigenvalues. Prove that if
$v_1+v_2+\ldots+v_k$ is in $W$, then $v_i\in W$ for all $i$.  Hint: Use induction on $k$.




     \end{enumerate}
