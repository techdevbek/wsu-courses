 defined by $T(A)=A^T$ (the transpose of $A$) is a linear transformation.  Let $\beta$ be the standard basis for $M_{2\times2}(\mathbb{F})$.
 \begin{enumerate}
   \item Find $A=[T]_\beta^\beta$
   \item Find $A^{-1}$
   \item Describe $T^{-1}$ as a linear transformation from $M_{2\times2}(\mathbb{F})\rightarrow M_{2\times2}(\mathbb{F})$
 \end{enumerate}




\item Let $V$ and $W$ be finite-dimensional vector spaces and $T:V\rightarrow W$ be an isomorphism. Let
$U$ be a subspace of $V$.
\begin{enumerate}
\item Prove that
$T(U)$ is a subspace of $W$.

\item Prove that
dim$(U)=$ dim$(T(U))$.

\end{enumerate}

Prove that $RQ=[I_V]_\alpha^\gamma$
\item Prove that $Q^{-1}=[I_V]_\beta^\alpha$.
\end{enumerate}

\medskip\noindent DEFINITION:
Let $V$ be a finite-dimensional vector space and $T:V\rightarrow V$ a linear transformation.  A subspace $W$ is said to be $T$-invariant provided $T(w)\in W\ \forall w\in W.$
\item Let $V$ be an $n$-dimensional vector space, $T:V\rightarrow V$ a linear transformation,  and $W$ a $T$-invariant subspace with dim$(W)=m<n$.  Prove that there is a basis $\beta$ for $V$ such that $[T]_\beta^\beta=\left[\begin{array}{cc}A&C\\0&B\end{array}\right]$, where $A$ is an $m\times m$ matrix, $B$ is an $(n-m)\times(n-m)$ matrix, and the zero block is $(n-m)\times m$.

     \end{enumerate}
