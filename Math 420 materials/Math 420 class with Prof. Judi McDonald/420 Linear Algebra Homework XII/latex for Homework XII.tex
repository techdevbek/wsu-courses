\begin{enumerate}

\item For each linear operator $T$ on the given inner product space $V$, determine whether $T$ is normal, self-adjoint, or neither. If possible, produce an orthonormal basis of eigenvectors of $T$ for $V$ and list the corresponding eigenvalues.
    \begin{enumerate}
    
    \item $V=\mathbb{C}^2$ and $T\left(\left[\begin{array}{c}
    a\\b\end{array}\right]\right)=\left[\begin{array}{c}
    2a+bi\\a+2b\end{array}\right]$
    \item $V=\mathbb{C}^2$ and $T\left(\left[\begin{array}{c}
    a\\b\end{array}\right]\right)=\left[\begin{array}{c}
    2a+bi\\-ai+2b\end{array}\right]$
    
    
    
    \item $V=M_{2\times 2}(\mathbb{R})$ and $T$ is defined by $T(A)=A^T$.

    \item $V=M_{2\times 2}(\mathbb{R})$ and $T$ is defined by $T\left(\left[\begin{array}{cc}
    a&b\\c&d\end{array}\right]\right)=\left[\begin{array}{cc}
    c&d\\a&b\end{array}\right]$
  
    \end{enumerate}

\item 
Let $T$ be a normal operator on a finite-dimensional complex inner product space $V$. Define
$$T_1=\frac{T+T*}{2}\mbox{ and } T_2=\frac{T-T*}{2i} $$
\begin{enumerate}
\item
Prove that $T_1$ and $T_2$ are self-adjoint and that $T=T_1+iT_2$.
\item Suppose also that $T=U_1+iU_2$
, where $U_1$
 and $U_2$
 are self-adjoint. Prove that $U_1=T_1$
 and $U_2=T_2$
.

\item Prove that $T$ is normal if and only if $T_1T_2=T_2T_1$.
\end{enumerate}

\item Let $T$ be a linear operator on an inner product space $V$, and let $W$ be a $T$-invariant subspace of $V$. Prove the following results.
\begin{enumerate}
\item
If $T$ is self-adjoint, then $T|_W$
 is self-adjoint.
\item $W^\perp$
 is $T^*$-invariant.
\item If $W$ is both $T$- and $T^*$-invariant, then $(T|_W)^*
=(T^*)|_W.$
\item
If $W$ is both $T$- and $T^*$-invariant and $T$ is normal, then $T|_W$
 is normal.
\end{enumerate}
\item
{\it Simultaneous Diagonalization:} Let $V$ be a finite-dimensional real inner product space, and let $U$ and $T$ be self-adjoint linear operators on $V$ such that $UT=TU$. Prove that there exists an orthonormal basis for $V$ consisting of vectors that are eigenvectors of both $U$ and $T$.
     \end{enumerate}
